function buildTicker(){
  const t=document.getElementById('ticker');
  const d=[...ticks,...ticks];
  t.innerHTML=d.map(i=>`<span class="tick"><span class="muted">${i.sym}</span><span>${i.price}</span><span class="${i.up?'up':'down'}">${i.chg}</span></span>`).join('');
}

function optFilter(type,btn){
  document.querySelectorAll('.filter-btn').forEach(b=>b.classList.remove('active'));
  btn.classList.add('active');
  document.querySelectorAll('.opt-card').forEach(c=>{
    if(type==='all') c.style.display='block';
    else if(type==='unusual') c.style.display=c.dataset.unusual==='true'?'block':'none';
    else c.style.display=c.dataset.type===type?'block':'none';
  });
}

function postGig(){
  const title=document.getElementById('gig-title-in').value;
  const price=document.getElementById('gig-price-in').value;
  const cat=document.getElementById('gig-cat-in').value;
  const desc=document.getElementById('gig-desc-in').value;
  if(!title||!price){toast('⚠️ Please fill in title and price');return;}
  const emojis={cleaning:'🧹',outdoor:'🌿',auto:'🚗',moving:'📦'};
  gigs.unshift({id:Date.now(),title,price,cat,desc:desc||'Quality service guaranteed.',name:'You',rating:'New',jobs:0,emoji:emojis[cat]||'🛠️'});
  document.getElementById('gig-list').innerHTML=gigs.map(gigCard).join('');
  document.getElementById('gig-form').style.display='none';
  document.getElementById('gig-title-in').value='';
  document.getElementById('gig-price-in').value='';
  document.getElementById('gig-desc-in').value='';
  toast('✅ Your gig is live!');
}

function findJobs(){
  const search=document.getElementById('job-search').value.toLowerCase();
  const type=document.getElementById('job-type').value;
  let results=jobsDB.filter(j=>
    (search===''||j.title.toLowerCase().includes(search)||j.desc.toLowerCase().includes(search))&&
    (type==='Full-time'||j.type===type||search!=='')
  );
  if(results.length===0) results=jobsDB;
  document.getElementById('job-results').innerHTML=results.map(j=>`
    <div class="job-card">
      <div class="job-title">${j.title}</div>
      <div class="job-company">${j.company}</div>
      <div class="job-meta">📍 ${j.location} · 💼 ${j.type}</div>
      <div class="job-salary">${j.salary}</div>
      <div style="font-size:12px;color:#666;margin-top:8px;margin-bottom:12px">${j.desc}</div>
      <div style="display:flex;gap:8px">
        <button class="btn btn-blue" style="margin-top:0;font-size:13px" onclick="toast('📨 Applying to ${j.title}...')">Quick Apply</button>
        <button class="btn btn-outline" style="margin-top:0;font-size:13px" onclick="toast('🔖 Job saved!')">Save</button>
      </div>
    </div>`).join('');
  toast(`Found ${results.length} opportunities`);
}
