function toggleJoin(id){
  const c=communities.find(c=>c.id===id);
  if(c){
    c.joined=!c.joined;
    c.members+=c.joined?1:-1;
    renderCommunities();
    toast(c.joined?`✅ Joined ${c.name}!`:`Left ${c.name}`);
  }
}

function likePost(id){
  const p=posts.find(p=>p.id===id);
  if(p){p.liked=!p.liked;p.likes+=p.liked?1:-1;renderFeed();}
}

function createPost(){
  const text=document.getElementById('post-text').value.trim();
  if(!text){toast('⚠️ Please write something first');return;}
  posts.unshift({id:Date.now(),name:'You',avatar:'YO',color:'var(--gold)',community:'Options Flow Traders',time:'Just now',body:text,likes:0,comments:0,liked:false});
  document.getElementById('post-text').value='';
  renderFeed();
  toast('✅ Posted to your community!');
}

function createCommunity(){
  const name=document.getElementById('comm-name').value;
  const cat=document.getElementById('comm-cat').value;
  const desc=document.getElementById('comm-desc').value;
  if(!name){toast('⚠️ Please enter a community name');return;}
  communities.unshift({id:Date.now(),name,cat,icon:catEmojis[cat]||'🌐',color:catColors[cat]||'#aaa',desc:desc||'A new community on EntangleWealth.',members:1,posts:0,joined:true});
  closeModal('create-community');
  renderCommunities();
  toast(`✅ ${name} community created!`);
}

function createEvent(){
  const title=document.getElementById('ev-title').value;
  const date=document.getElementById('ev-date').value;
  const location=document.getElementById('ev-location').value;
  const type=document.getElementById('ev-type').value;
  const desc=document.getElementById('ev-desc').value;
  const price=document.getElementById('ev-price').value;
  if(!title||!date){toast('⚠️ Please fill in title and date');return;}
  events.unshift({id:Date.now(),title,date:new Date(date).toLocaleDateString('en-US',{month:'short',day:'numeric',year:'numeric'}),location:location||'TBD',type,desc:desc||'Join us for this exciting event.',price:price==='0'||price===''?'Free':price,attendees:1,virtual:location.toLowerCase().includes('virtual')||location.toLowerCase().includes('zoom')});
  closeModal('create-event');
  renderEvents();
  toast(`✅ ${title} created!`);
}
